Given a solidity function enableTokenTransfer and the related parts from a smart contract.
Whether the function enableTokenTransfer and the related parts violate the rule? Replay in "YES" or "NO".
Notes: Before Solidity version 0.5, events are emitting without "emit" keyword.

Example of "initial token distribution" are provided below. 
Example:"""
construct <Contract Name> {
    constructor() {
        // tokens are transfer during the initial token distribution:
        balances[msg.sender] = total;
    }
}
"""



Function(and related parts) to audit:"""
pragma solidity 0.4.21;
contract Owner {

    address public owner;
    modifier onlyOwner() {
        require(msg.sender == owner);
        _;
    }
}

contract KIMEX is Owner {
    bool public inActive;
    bool public isTransferable;

    modifier isActive() {
        require(inActive == false);
        _;
    }

    // Enable transfer feature of tokens
    function enableTokenTransfer() external isActive onlyOwner {
        isTransferable = true;
    }
}
"""

Rule:"""
Emit Transfer if tokens are transferred during the initial token distribution or crowdsale.
