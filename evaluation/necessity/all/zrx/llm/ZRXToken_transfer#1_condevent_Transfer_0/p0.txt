contract StandardToken is Token {
    function transfer(address _to, uint _value) returns (bool) {
        //Default assumes totalSupply can't be over max (2^256 - 1).
        if (
            balances[msg.sender] >= _value &&
            balances[_to] + _value >= balances[_to]
        ) {
            balances[msg.sender] -= _value;
            balances[_to] += _value;

            return true;
        } else {
            return false;
        }
    }

}
